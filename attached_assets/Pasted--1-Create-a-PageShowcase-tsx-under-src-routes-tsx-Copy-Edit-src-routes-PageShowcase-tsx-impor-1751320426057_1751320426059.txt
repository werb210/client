// src/routes/PageShowcase.tsx
import { Link } from "wouter";

const dashboards = [
  { name: "✅ AdminDashboard (correct one)", path: "/_show/admin" },
  { name: "❌ Legacy Dashboard",             path: "/_show/legacy" },
  { name: "📊 Staff Dashboard",              path: "/_show/staff" },
  { name: "📈 Marketing Dashboard",          path: "/_show/marketing" },
  { name: "🏦 Lender Dashboard",             path: "/_show/lender" },
  { name: "📩 Referrer Dashboard",           path: "/_show/referrer" },
  { name: "🧪 Misc Test Pages",              path: "/_show/test" }
];

export default function PageShowcase() {
  return (
    <div className="p-6 space-y-4">
      <h1 className="text-2xl font-bold">Page Showcase Dashboard</h1>
      <ul className="space-y-2">
        {dashboards.map(d => (
          <li key={d.path}>
            <Link href={d.path}>
              <button className="bg-blue-600 text-white px-4 py-2 rounded hover:bg-blue-700 shadow">
                {d.name}
              </button>
            </Link>
          </li>
        ))}
      </ul>
    </div>
  );
}
