// scripts/clientDiagnosticSuite.ts
export const runTests = async () => {
  const base = import.meta.env.VITE_API_BASE_URL;

  const results = {
    health: await test(`${base}/health`),
    login: await test(`${base}/auth/login`),
    register: await test(`${base}/auth/register`),
    reset: await test(`${base}/auth/request-reset`)
  };

  return {
    status: Object.values(results).every(r => r.success) ? "✅ PASSED" : "❌ FAILED",
    results
  };
};

const test = async (url: string) => {
  try {
    const res = await fetch(url, { method: "OPTIONS", mode: "cors" });
    const contentType = res.headers.get("content-type");
    return {
      url,
      status: res.status,
      success: res.ok && contentType?.includes("application/json"),
      contentType,
    };
  } catch (e) {
    return { url, success: false, error: e.message };
  }
};
